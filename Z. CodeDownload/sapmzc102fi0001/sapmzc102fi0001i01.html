MODULE exit_100 INPUT.

  CASE gv_okcode.
    WHEN 'BACK' OR 'EXIT' OR 'CANC'.
      CALL METHOD : go_bottom_alv_grid->free,
                    go_bottom_cont->free,
*--------------------------------------------------------------------*
                    go_top_alv_grid->free,
                    go_top_cont->free,
*--------------------------------------------------------------------*
                    go_split_cont->free,
                    go_base_container->free.

      FREE : go_bottom_alv_grid, go_bottom_cont,
             go_top_alv_grid, go_top_cont,
             go_split_cont, go_base_container.

      LEAVE TO SCREEN 0.
  ENDCASE.

ENDMODULE.

MODULE user_command_0100 INPUT.

  CASE gv_okcode.
    WHEN 'SRCH'.
      CASE 'X'.
        WHEN rb1.
          PERFORM show_all_alv_data. " 전체 데이터
        WHEN rb2.
          PERFORM show_ap_alv_data.  " 매입채무 관련 데이터 AR
        WHEN rb3.
          PERFORM show_ar_alv_data.  " 매출채권 관련 데이터 AP
      ENDCASE.
    WHEN 'RFSH'.
      PERFORM clear_screen. " 새로고침 시 화면 깨끗
  ENDCASE.

ENDMODULE.
