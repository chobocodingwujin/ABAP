FUNCTION ZFM_CL1_05_03.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     REFERENCE(IV_MATNR) TYPE  MATNR
*"     REFERENCE(IV_LANGU) TYPE  SPRAS
*"  EXPORTING
*"     REFERENCE(EV_MAKTX) TYPE  MAKTX
*"     REFERENCE(EV_MESSAGE) TYPE  CHAR50
*"----------------------------------------------------------------------

*       Global data declarations

SELECT SINGLE maktx
  FROM makt
  INTO ev_maktx
  WHERE spras EQ iv_langu
    AND matnr EQ iv_matnr.

IF ev_maktx IS INITIAL.
  ev_message = TEXT-e02.
ELSE.
  ev_message = TEXT-s01.

ENDIF.



ENDFUNCTION.
