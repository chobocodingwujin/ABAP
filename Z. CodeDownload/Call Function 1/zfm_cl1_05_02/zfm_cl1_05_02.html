FUNCTION zfm_cl1_05_02.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     REFERENCE(IV_HKONT) TYPE  HKONT
*"     REFERENCE(IV_KTOPL) TYPE  KTOPL
*"     REFERENCE(IV_LANGU) TYPE  SY-LANGU DEFAULT 'E'
*"  EXPORTING
*"     REFERENCE(EV_TXT50) TYPE  TXT50
*"     REFERENCE(EV_MESSAGE) TYPE  CHAR50
*"     REFERENCE(EV_SUBRC) TYPE  SY-SUBRC
*"  EXCEPTIONS
*"      NOT_FOUND
*"----------------------------------------------------------------------

*       Global data declarations

*-- Get G/L Account name
  SELECT SINGLE txt50
    FROM skat
    INTO ( ev_txt50 )
    WHERE spras EQ iv_langu
      AND saknr EQ iv_hkont
      AND ktopl EQ iv_ktopl.

*-- ELSEIF는 뒤에 뭐가 있어야함
*-- Set result data

  ev_subrc = sy-subrc.

  IF ev_subrc NE 0.
    ev_message = TEXT-e01.
    RAISE not_found.

  ENDIF.


ENDFUNCTION.
