FUNCTION zfm_cl1_05_04.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     REFERENCE(IV_STLAN) TYPE  STLAN
*"     REFERENCE(IV_LANGU) TYPE  SPRAS
*"  EXPORTING
*"     REFERENCE(EV_ANTXT) TYPE  ANTXT
*"     REFERENCE(EV_MESSAGE) TYPE  CHAR50
*"----------------------------------------------------------------------

*       Global data declarations
  SELECT SINGLE antxt
    FROM t416t
    INTO ev_antxt
    WHERE stlan EQ iv_stlan
      AND spras EQ iv_langu.

  IF ev_antxt IS INITIAL.
    ev_message = TEXT-e03.
  ELSE.
    ev_message = TEXT-s01.
  ENDIF.




ENDFUNCTION.
