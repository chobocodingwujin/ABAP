FUNCTION zfm_cl102_fi_02.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     REFERENCE(IV_ANLN1) TYPE  ZC102FIT0007-ANLN1
*"     REFERENCE(IV_DESUM) TYPE  ZC102FIT0012-DESUM
*"     REFERENCE(IV_DEBIT_WRBTR_COST) TYPE  ZC102FIT0010-WRBTR
*"     REFERENCE(IV_CREDIT_WRBTR_ASSET) TYPE  ZC102FIT0010-WRBTR
*"     REFERENCE(IV_DEBIT_WRBTR_CASH) TYPE  ZC102FIT0010-WRBTR
*"  EXPORTING
*"     REFERENCE(EV_NUM) TYPE  ZC102FIT0009-BELNR
*"----------------------------------------------------------------------
*-- For Setting
  DATA : lt_bschl TYPE TABLE OF zc102fit0004,
         ls_bschl TYPE zc102fit0004.

  DATA : lv_buzei TYPE zc102fit0010-buzei.

*-- For Save
  DATA : lt_header TYPE TABLE OF zc102fit0009,
         ls_header TYPE zc102fit0009,
         lt_line   TYPE TABLE OF zc102fit0010,
         ls_line   TYPE zc102fit0010,
         ls_table  TYPE zc102mmt0015.

*-- 전기키 마스터
  CLEAR : lt_bschl, ls_bschl, lv_buzei.

  SELECT bschl shkzg koart
    INTO CORRESPONDING FIELDS OF TABLE lt_bschl
    FROM zc102fit0004.

*-- Set Document Number
  CALL FUNCTION 'NUMBER_GET_NEXT'
    EXPORTING
      nr_range_nr = '01'
      object      = 'ZC102_FI'
    IMPORTING
      number      = ev_num.

  ls_header = VALUE #( bukrs = '1000'
                       belnr = ev_num
                       gjahr = sy-datum
                       blart = 'SA'
                       budat = sy-datum
                       bldat = sy-datum
                       bktxt = '자산 처분 손실'
                       waers = 'KRW'
                       erdat = sy-datum
                       ernam = sy-uname
                       erzet = sy-uzeit ).
  APPEND ls_header TO lt_header.

*-- Set Line Item
  CLEAR ls_line.
  ls_line-bschl = '40'.     " 일반 차변
  ls_line-saknr = '190000'. " 감가상각누계액
  ls_line-wrbtr = iv_desum.
  ls_line-sgtxt = '감가상각누계액'.

  APPEND ls_line TO lt_line.

  CLEAR ls_line.
  ls_line-bschl = '40'.     " 일반 차변
  ls_line-saknr = '100002'. " 당좌예금
  ls_line-wrbtr = iv_debit_wrbtr_cash.
  ls_line-sgtxt = '처분가액'.

  APPEND ls_line TO lt_line.

  CLEAR ls_line.
  ls_line-bschl = '40'.     " 일반 차변
  ls_line-saknr = '500004'. " 유형 자산 처분 손실
  ls_line-wrbtr = iv_debit_wrbtr_cost.
  ls_line-sgtxt = '유형자산 처분 손실'.

  APPEND ls_line TO lt_line.

  CLEAR ls_line.
  ls_line-bschl = '50'.     " 일반 대변
  ls_line-saknr = '160000'. " 유형 자산
  ls_line-wrbtr = iv_credit_wrbtr_asset.
  ls_line-sgtxt = '유형자산 취득가액'.
  ls_line-anln1 = iv_anln1.

  APPEND ls_line TO lt_line.

*-- Data Setting
  LOOP AT lt_line INTO ls_line.
    lv_buzei += 1.
    ls_line = VALUE #( BASE ls_line
                       bukrs = ls_header-bukrs
                       belnr = ls_header-belnr
                       gjahr = ls_header-gjahr
                       buzei = lv_buzei
                       shkzg = ls_bschl-shkzg
                       koart = ls_bschl-koart
                       kokrs = '1000'
*                         mwskz =
                       waers = ls_header-waers
                       ernam = sy-uname
                       erdat = sy-datum
                       erzet = sy-uzeit ).
    READ TABLE lt_bschl INTO ls_bschl WITH KEY bschl = ls_line-bschl.

    IF sy-subrc = 0.
      ls_line = VALUE #( BASE ls_line shkzg = ls_bschl-shkzg
                                      koart = ls_bschl-koart ).
      MODIFY lt_line FROM ls_line.
    ENDIF.
  ENDLOOP.

  MODIFY zc102fit0009 FROM TABLE lt_header.
  MODIFY zc102fit0010 FROM TABLE lt_line.
ENDFUNCTION.
